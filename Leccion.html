📊 Calculadora de IMC

Aplicación en Python con interfaz gráfica usando Tkinter

📘 ¿Qué es el IMC?

El Índice de Masa Corporal (IMC) es una medida que permite evaluar si una persona tiene un peso adecuado en relación con su estatura.

🧠 Fórmula del IMC


IMC = peso / (estatura)^2

📊 Clasificación

Bajo peso: menor a 18.5
Normal: 18.5 – 24.9
Sobrepeso: 25 – 29.9
Obesidad: 30 o más

💻 Implementación en Python


imc = peso / (estatura ** 2)

import tkinter as tk
from tkinter import messagebox

# =========================================================
# FUNCIÓN: animar_barra
# =========================================================
def animar_barra(actual, objetivo, color):
    """
    Esta función se encarga de animar la barra de progreso.
    
    Parámetros:
    - actual: posición actual de la barra (ancho actual)
    - objetivo: ancho final que debe alcanzar la barra
    - color: color que representa el estado del IMC
    """

    # Verifica si la barra aún no ha llegado al objetivo
    if actual < objetivo:
        actual += 2  # Incremento del ancho (controla la velocidad)

        # Actualiza las coordenadas de la barra (x1, y1, x2, y2)
        canvas_barra.coords(barra, 0, 0, actual, 20)

        # Cambia el color de la barra dinámicamente
        canvas_barra.itemconfig(barra, fill=color)

        # Llama nuevamente a la función después de 10 ms (animación)
        ventana.after(10, animar_barra, actual, objetivo, color)


# =========================================================
# FUNCIÓN: calcular_imc
# =========================================================
def calcular_imc():
    """
    Esta función:
    1. Obtiene los datos ingresados por el usuario
    2. Calcula el Índice de Masa Corporal (IMC)
    3. Determina el estado de salud según el resultado
    4. Muestra el resultado en pantalla
    5. Actualiza la barra de progreso con animación
    """

    try:
        # ==============================
        # OBTENCIÓN DE DATOS
        # ==============================

        nombre = entry_nombre.get()  # Texto (string)
        edad = int(entry_edad.get())  # Conversión a entero
        peso = float(entry_peso.get())  # Conversión a decimal
        estatura = float(entry_estatura.get())  # En metros

        # ==============================
        # VALIDACIÓN DE DATOS
        # ==============================

        # Verifica que el nombre no esté vacío
        if nombre == "":
            messagebox.showwarning("Error", "Completa todos los campos")
            return  # Detiene la ejecución

        # ==============================
        # CÁLCULO DEL IMC
        # Fórmula: peso / (estatura^2)
        # ==============================

        imc = peso / (estatura ** 2)

        # ==============================
        # CLASIFICACIÓN DEL IMC
        # Se asigna un estado y color
        # ==============================

        if imc < 18.5:
            estado = "Bajo peso"
            color = "#facc15"  # Amarillo
        elif imc < 24.9:
            estado = "Normal"
            color = "#22c55e"  # Verde
        elif imc < 29.9:
            estado = "Sobrepeso"
            color = "#fb923c"  # Naranja
        else:
            estado = "Obesidad"
            color = "#ef4444"  # Rojo

        # ==============================
        # MOSTRAR RESULTADO
        # ==============================

        resultado.set(f"{nombre}, tu IMC es {imc:.2f}\nEstado: {estado}")

        # Cambia el color del texto según el resultado
        label_resultado.config(fg=color)

        # ==============================
        # BARRA DE PROGRESO
        # ==============================

        # Se normaliza el IMC a un rango de 0 a 1
        # Se toma 40 como valor máximo de referencia
        porcentaje = min(imc / 40, 1)

        # Convierte el porcentaje en píxeles (ancho de la barra)
        objetivo = int(porcentaje * 300)

        # Reinicia la barra a 0 antes de animar
        canvas_barra.coords(barra, 0, 0, 0, 20)

        # Llama a la función de animación
        animar_barra(0, objetivo, color)

    except ValueError:
        # Manejo de error si el usuario ingresa texto en lugar de números
        messagebox.showerror("Error", "Ingresa valores válidos")


# =========================================================
# CREACIÓN DE LA VENTANA PRINCIPAL
# =========================================================

ventana = tk.Tk()
ventana.title("Calculadora de IMC")
ventana.geometry("400x500")

# Color de fondo de la ventana (tema oscuro)
ventana.config(bg="#0f172a")


# =========================================================
# TÍTULO DE LA APLICACIÓN
# =========================================================

titulo = tk.Label(
    ventana,
    text="Calculadora de IMC",
    font=("Poppins", 18, "bold"),
    bg="#0f172a",
    fg="#38bdf8"
)
titulo.pack(pady=10)


# =========================================================
# CONTENEDOR PRINCIPAL (FRAME)
# =========================================================

frame = tk.Frame(ventana, bg="#1e293b")
frame.pack(padx=20, pady=20, fill="both", expand=True)


# =========================================================
# CAMPOS DE ENTRADA
# =========================================================

# Campo: Nombre
tk.Label(frame, text="Nombre:", bg="#1e293b", fg="white").pack(anchor="w", padx=10)
entry_nombre = tk.Entry(frame)
entry_nombre.pack(padx=10, pady=5, fill="x")

# Campo: Edad
tk.Label(frame, text="Edad:", bg="#1e293b", fg="white").pack(anchor="w", padx=10)
entry_edad = tk.Entry(frame)
entry_edad.pack(padx=10, pady=5, fill="x")

# Campo: Peso
tk.Label(frame, text="Peso (kg):", bg="#1e293b", fg="white").pack(anchor="w", padx=10)
entry_peso = tk.Entry(frame)
entry_peso.pack(padx=10, pady=5, fill="x")

# Campo: Estatura
tk.Label(frame, text="Estatura (m):", bg="#1e293b", fg="white").pack(anchor="w", padx=10)
entry_estatura = tk.Entry(frame)
entry_estatura.pack(padx=10, pady=5, fill="x")


# =========================================================
# BOTÓN PARA CALCULAR EL IMC
# =========================================================

btn_calcular = tk.Button(
    frame,
    text="Calcular IMC",
    command=calcular_imc,  # Llama a la función principal
    bg="#0ea5e9",
    fg="white",
    font=("Poppins", 10, "bold"),
    relief="flat"
)
btn_calcular.pack(pady=15)


# =========================================================
# ETIQUETA PARA MOSTRAR RESULTADOS
# =========================================================

resultado = tk.StringVar()  # Variable dinámica para mostrar texto

label_resultado = tk.Label(
    frame,
    textvariable=resultado,
    bg="#1e293b",
    fg="#22d3ee",
    font=("Arial", 12, "bold"),
    justify="center"
)
label_resultado.pack(pady=10)


# =========================================================
# BARRA DE PROGRESO (CANVAS)
# =========================================================

tk.Label(frame, text="Nivel IMC:", bg="#1e293b", fg="white").pack()

canvas_barra = tk.Canvas(
    frame,
    width=300,
    height=20,
    bg="#334155",
    highlightthickness=0
)
canvas_barra.pack(pady=10)

# Se crea un rectángulo que representa la barra (inicialmente vacío)
barra = canvas_barra.create_rectangle(0, 0, 0, 20, fill="#22c55e")


# =========================================================
# EJECUCIÓN DE LA APLICACIÓN
# =========================================================

# Mantiene la ventana abierta en ejecución
ventana.mainloop()
