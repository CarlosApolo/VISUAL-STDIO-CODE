import tkinter as tk
from tkinter import messagebox
import json, os

ARCHIVO = "finanzas.json"

balance = 0
historial = []
ingresos_total = 0
gastos_total = 0

# -------- FUNCIONES --------

def centrar_ventana(ventana, ancho=420, alto=650):
    pantalla_ancho = ventana.winfo_screenwidth()
    pantalla_alto = ventana.winfo_screenheight()
    x = int((pantalla_ancho / 2) - (ancho / 2))
    y = int((pantalla_alto / 2) - (alto / 2))
    ventana.geometry(f"{ancho}x{alto}+{x}+{y}")

def cargar_datos():
    global balance, historial, ingresos_total, gastos_total
    if os.path.exists(ARCHIVO):
        with open(ARCHIVO, "r") as f:
            datos = json.load(f)
            balance = datos.get("balance", 0)
            historial = datos.get("historial", [])
            ingresos_total = datos.get("ingresos", 0)
            gastos_total = datos.get("gastos", 0)
            actualizar()

def guardar_datos():
    datos = {
        "balance": balance,
        "historial": historial,
        "ingresos": ingresos_total,
        "gastos": gastos_total
    }
    with open(ARCHIVO, "w") as f:
        json.dump(datos, f)

def agregar_ingreso():
    global balance, ingresos_total
    try:
        monto = float(entrada_monto.get())
        desc = entrada_desc.get()

        balance += monto
        ingresos_total += monto
        historial.append(f"➕ {desc}: ${monto}")

        animar_balance("#00ff9c")
        guardar_datos()
        actualizar()
        limpiar()
    except:
        messagebox.showerror("Error", "Monto inválido")

def agregar_gasto():
    global balance, gastos_total
    try:
        monto = float(entrada_monto.get())
        desc = entrada_desc.get()

        balance -= monto
        gastos_total += monto
        historial.append(f"➖ {desc}: ${monto}")

        animar_balance("#ff4d6d")
        guardar_datos()
        actualizar()
        limpiar()
    except:
        messagebox.showerror("Error", "Monto inválido")

def actualizar():
    lista.delete(0, tk.END)
    for item in historial:
        lista.insert(tk.END, item)
    etiqueta_balance.config(text=f"${balance:.2f}")
    actualizar_barra()

def limpiar():
    entrada_desc.delete(0, tk.END)
    entrada_monto.delete(0, tk.END)

# -------- ANIMACIÓN --------

def animar_balance(color):
    etiqueta_balance.config(fg=color)
    ventana.after(300, lambda: etiqueta_balance.config(fg="white"))

# -------- BARRA VISUAL --------

def actualizar_barra():
    total = ingresos_total + gastos_total
    canvas.delete("all")

    if total == 0:
        return

    ingreso_width = (ingresos_total / total) * 300
    gasto_width = (gastos_total / total) * 300

    canvas.create_rectangle(0, 0, ingreso_width, 30, fill="#00ff9c")
    canvas.create_rectangle(ingreso_width, 0, ingreso_width + gasto_width, 30, fill="#ff4d6d")

# -------- RESET --------

def reiniciar_datos():
    global balance, historial, ingresos_total, gastos_total

    confirmar1 = messagebox.askyesno(
        "Confirmación",
        "¿Seguro que deseas eliminar TODOS los datos?"
    )

    if confirmar1:
        confirmar2 = messagebox.askyesno(
            "Confirmación final",
            "Esta acción es irreversible.\n¿Deseas continuar?"
        )

        if confirmar2:
            balance = 0
            historial = []
            ingresos_total = 0
            gastos_total = 0

            guardar_datos()
            actualizar()

            messagebox.showinfo("Listo", "Datos reiniciados correctamente")

# -------- HOVER --------

def hover_on(e):
    e.widget["bg"] = "#00c37a"

def hover_off(e):
    e.widget["bg"] = "#00ff9c"

def hover_reset_on(e):
    e.widget["bg"] = "#d50000"

def hover_reset_off(e):
    e.widget["bg"] = "#ff1744"

# -------- VENTANA --------

ventana = tk.Tk()
ventana.title("Finanzas PRO")
ventana.configure(bg="#0f172a")

centrar_ventana(ventana)

# -------- HEADER --------

header = tk.Frame(ventana, bg="#1e293b")
header.pack(fill="x", pady=10, padx=10)

tk.Label(header, text="💰 Finanzas PRO",
         font=("Arial", 18, "bold"),
         fg="#38bdf8", bg="#1e293b").pack()

tk.Label(header, text="Gestiona tu dinero fácilmente",
         fg="#cbd5f5", bg="#1e293b").pack()

# -------- BALANCE --------

card = tk.Frame(ventana, bg="#1e293b")
card.pack(padx=15, pady=(10, 2), fill="x")

tk.Label(card, text="Balance", fg="#94a3b8", bg="#1e293b").pack()

etiqueta_balance = tk.Label(card, text="$0.00",
                            font=("Arial", 22, "bold"),
                            fg="white", bg="#1e293b")
etiqueta_balance.pack()

# -------- BARRA --------

canvas = tk.Canvas(ventana, width=300, height=30,
                   bg="#0f172a", highlightthickness=0)
canvas.pack(pady=10)

# -------- FORM --------

form = tk.Frame(ventana, bg="#0f172a")
form.pack(padx=15, pady=(2, 10), fill="x")

# -------- DESCRIPCIÓN --------
tk.Label(form,
         text="Descripción",
         fg="#cbd5f5",
         bg="#0f172a",
         font=("Arial", 10, "bold")
).pack(anchor="w", pady=(5, 0))

entrada_desc = tk.Entry(
    form,
    bg="#1e293b",
    fg="white",
    insertbackground="white",
    relief="flat"
)
entrada_desc.pack(fill="x", pady=5, ipady=5)


# -------- MONTO --------
tk.Label(form,
         text="Monto ($)",
         fg="#cbd5f5",
         bg="#0f172a",
         font=("Arial", 10, "bold")
).pack(anchor="w", pady=(10, 0))

entrada_monto = tk.Entry(
    form,
    bg="#1e293b",
    fg="white",
    insertbackground="white",
    relief="flat"
)
entrada_monto.pack(fill="x", pady=5, ipady=5)

# -------- BOTONES --------

botones = tk.Frame(ventana, bg="#0f172a")
botones.pack(pady=10)

btn_ingreso = tk.Button(botones, text="➕ Ingreso",
                        bg="#00ff9c", fg="black", width=15,
                        relief="flat", command=agregar_ingreso)
btn_ingreso.grid(row=0, column=0, padx=5)

btn_ingreso.bind("", hover_on)
btn_ingreso.bind("", hover_off)

btn_gasto = tk.Button(botones, text="➖ Gasto",
                      bg="#ff4d6d", fg="white", width=15,
                      relief="flat", command=agregar_gasto)
btn_gasto.grid(row=0, column=1, padx=5)

# -------- BOTÓN RESET --------

btn_reset = tk.Button(ventana,
                      text="🗑️ Reiniciar datos",
                      bg="#ff1744",
                      fg="white",
                      relief="flat",
                      command=reiniciar_datos)
btn_reset.pack(pady=10)

btn_reset.bind("", hover_reset_on)
btn_reset.bind("", hover_reset_off)

# -------- HISTORIAL --------

frame_historial = tk.Frame(ventana, bg="#1e293b")
frame_historial.pack(padx=15, pady=10, fill="both", expand=True)

tk.Label(frame_historial, text="Historial",
         fg="white", bg="#1e293b",
         font=("Arial", 11, "bold")).pack(anchor="w")

scrollbar = tk.Scrollbar(frame_historial)
scrollbar.pack(side="right", fill="y")

lista = tk.Listbox(
    frame_historial,
    bg="#020617",
    fg="white",
    selectbackground="#38bdf8",
    relief="flat",
    yscrollcommand=scrollbar.set
)
lista.pack(side="left", fill="both", expand=True)

scrollbar.config(command=lista.yview)

# -------- INICIO --------

cargar_datos()

ventana.mainloop()
